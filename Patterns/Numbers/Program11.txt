class Pattern11 {
	public static void main(String [] args){
		int n = 5;
		int count = n;
		for(int i = 1; i<=n;i++){
			int c= count;
			for(int j=1;j<=i;j++){
				System.out.print(c+"\t");
				c -=n-j+i;
			}
			System.out.println();
			count+=n-i;
		}
	}
}
